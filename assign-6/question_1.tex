

\newtheorem{theorem}{Theorem}

\begin{theorem}
  If $A$ is a set and $\{B_i | i \in I\}$ is an indexed family of sets. $A
  \times (\cup_{i\in I}=\cup_{i\in I})(A \times B)$.
\end{theorem}

\begin{proof}
  Suppose $(a,b) \in A \times (\cup_{i\in I})$. Then $a \in A$ and $b \in
  \cup_{i\in I}$.
\end{proof}
